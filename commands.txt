## set up new docker container with new postgres database
docker run -p 5432:5432 -d \
  -e POSTGRES_PASSWORD=postgres \
  -e POSTGRES_USER=postgres \
  -e POSTGRES_DB=sdc-qa \
  -v pgdata:/var/lib/postgresql/data \
  postgres

## accessing my own local database
psql sdc -h localhost

docker exec -it 2d14079a8fec psql -U postgres sdc-qa
